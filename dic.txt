9: Flate Dictionary.

flat = {}
data = {
    "name": "Shankar",
    "details": {
        "age": 25,
        "address": {
            "city": "Delhi",
            "pincode": 110001
        }
    }
}

def flatten_dict(d, parent_key=''):
    for key, value in d.items():
        new_key = f"{parent_key}.{key}" if parent_key else key

        if isinstance(value, dict):  # agar value dict hai
            flatten_dict(value, new_key)  # recursion
        else:
            flat[new_key] = value


flatten_dict(data)
print(flat)
