1: Salary above average.

SELECT *
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);


avg_salary = Employee.average(:salary)

Employee.where("salary > ?", avg_salary)


2: Customers having more than 1 order.

SELECT customer_id
FROM orders
GROUP BY customer_id
HAVING COUNT(*) > 1;


Customer.joins(:orders)
        .group(:id)
        .having("COUNT(orders.id) > 1")
        

Customer.joins(:orders)
        .select(
          "customers.id,
           customers.name,
           SUM(orders.amount) AS total_orders"
        )
        .group(
          "customers.id,
           customers.name"
        )
        .having(
          "COUNT(orders.id) > 1"
        )



3: Department wise employee count.

SELECT department,
       COUNT(*) AS total_emp
FROM employees
GROUP BY department
ORDER BY total_emp ASC;

Employee.group(:department)
        .order("COUNT(*) ASC")
        .count


4: Duplicate salaries.

SELECT salary
FROM employees
GROUP BY salary
HAVING COUNT(*) > 1;


Employee.group(:salary)
        .having("COUNT(*) > 1")
        .count
        
        
5: Employees without manager.

SELECT *
FROM employees
WHERE manager_id IS NULL;


Employee.where(manager_id: nil)


6: Employees hired in last 3 months.

SELECT *
FROM employees
WHERE hire_date >= DATE_SUB(CURDATE(), INTERVAL 3 MONTH);


Employee.where(
  "hire_date >= ?",
  3.months.ago.to_date
)


7: Name contains "Shankar".

SELECT *
FROM employees

Employee.where(
  "name LIKE ?",
  "%Shankar%"
)


8: Customers with more than 1 order.


SELECT c.*,
       COUNT(o.id) AS total_orders
FROM customers c
LEFT JOIN orders o
ON c.id = o.customer_id
GROUP BY c.id
HAVING COUNT(o.id) > 1;


Customer.left_joins(:orders)
        .group(:id)
        .having("COUNT(orders.id) > 1")
        
        
9: Order ID 1 products and quantities.

SELECT p.product_name,
       oi.quantity
FROM order_items oi
JOIN products p
ON oi.product_id = p.id
WHERE oi.order_id = 1;


OrderItem.joins(:product)
         .where(order_id: 1)
         .pluck(
           "products.product_name",
           :quantity
         )

10: Customers who have not placed an order.


SELECT c.*
FROM customers c
LEFT JOIN orders o
ON c.id = o.customer_id
WHERE o.id IS NULL;



Customer.left_joins(:orders)
        .where(orders: { id: nil })
        
        




11: Retrieve the names of the top 10 users who have made the most
orders, along with the count of their orders.


SELECT users.*,
       COUNT(orders.id) AS total_orders
FROM users
LEFT JOIN orders
ON users.id = orders.user_id
GROUP BY users.id
ORDER BY total_orders DESC
LIMIT 10;


User.left_joins(:orders)
    .select("users.*, COUNT(orders.id) AS total_orders")
    .group("users.id")
    .order("total_orders DESC")
    .limit(10)
