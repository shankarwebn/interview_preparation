7: Palindrome number.

num = int(input("Enter the number "))
tem = num
rev = 0

while(num>0):
    d = num%10
    rev = rev * 10 + d
    num = num//10

if tem ==rev:
    print("Palindrome")
else:
    print("Not a palindrome")

11: GCD of two numbers.

num1 = int(input("Enter the first number: "))
num2 = int(input("Enter the second number: "))

c = num1 if num1 < num2 else num2
a = 1
gcd_list = []

while a <= c:  # <= important
    if num1 % a == 0 and num2 % a == 0:
        gcd_list.append(a)
    a += 1       # increment ab loop ke bahar hai, har step chalega

print("Common divisors:", gcd_list)
print("GCD:", max(gcd_list))

13: Fibonacci using recursion.

def fibonacci(n):
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci(n-1) + fibonacci(n-2)

n = int(input("Enter number of terms: "))

for i in range(n):
    print(fibonacci(i), end=" ")
