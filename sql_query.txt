14: Top 5 Highest Salary.
	
SELECT emp_id, emp_name, salary
FROM (
    SELECT *, DENSE_RANK() OVER (ORDER BY salary DESC) AS rn
    FROM employees
) AS t
WHERE rn <= 5;
