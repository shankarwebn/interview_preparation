SELECT *

WHERE salary > (SELECT AVG(salary) FROM employees);

13: Nth highest salary.

SELECT * from ( SELECT salary , DENSE_RANK() OVER (ORDER BY salary DESC)
as emm_rank from emp)t  WHERE emm_rank = 2;



from django.db.models import F, Window
from django.db.models.functions import DenseRank

employees = (
    Employee.objects
    .annotate(
        emp_rank=Window(
            expression=DenseRank(),
            order_by=F('salary').desc()
        )
    )
    .filter(emp_rank=2)
)

14: Max salary for each Department;

FROM employees

Employee.objects.values('department').annotate(

)


from django.db.models import F, Max, Window

employees = (
    Employee.objects
    .annotate(
        max_salary=Window(

            partition_by=[F('department')]
        )
    )
    .filter(salary=F('max_salary'))
)

16: DELETe duplicate data.

DELETE FROM Employee
WHERE id NOT IN (
    SELECT min_id
    FROM (
        SELECT MIN(id) AS min_id
        FROM Employee
        GROUP BY name, department
    ) t
);



from django.db.models import Count, Min

duplicates = (
    Employee.objects
    .values('name', 'department')
    .annotate(
        min_id=Min('id'),
        total=Count('id')
    )
    .filter(total__gt=1)
)

for dup in duplicates:
    Employee.objects.filter(
        name=dup['name'],
        department=dup['department']
    ).exclude(
        id=dup['min_id']
    ).delete()
